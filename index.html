02

تمثيل الـ Graph

بدل مكتبات جاهزة، استخدمنا Arrays ثنائية البعد لتمثيل Adjacency List.

const int MAX = 100;

Course courses[MAX];
int courseCount = 0;

int prereq[MAX][MAX];
int prereqCount[MAX];

int adj[MAX][MAX];
int adjCount[MAX];
03

إضافة مادة جديدة

نقرأ بيانات المادة ونخزنها في المصفوفة، ثم نزيد العداد.

void addCourse() {
    Course c;
    cin >> c.id;

    getline(cin, c.name);
    cin >> c.hours;

    courses[courseCount] = c;
    courseCount++;
}
04

إضافة متطلب سابق

نخزن العلاقة في كلا الاتجاهين: المتطلب في prereq، والمادة التابعة في adj.

prereq[cIdx][prereqCount[cIdx]] = preId;
prereqCount[cIdx]++;

adj[pIdx][adjCount[pIdx]] = courseId;
adjCount[pIdx]++;
05

عرض المتطلبات (DFS)

نستخدم DFS recursive لنطبع المتطلبات المباشرة وغير المباشرة.

void showPrerequisites(int id) {
    int idx = courseIndex(id);
    if (idx == -1) return;

    for (int i = 0; i < prereqCount[idx]; i++) {
        int preId = prereq[idx][i];
        cout << "- " << preId << "\n";
        showPrerequisites(preId);
    }
}
06

كشف الحلقات

نستخدم DFS مع ثلاث حالات: 0 = ما زرته، 1 = قاعد أزوره، 2 = خلصت منه.

bool hasCycleDFS(int u, int state[]) {
    state[u] = 1;

    for (int i = 0; i < adjCount[u]; i++) {
        int v = adj[u][i];
        int vIdx = courseIndex(v);
        if (vIdx == -1) continue;

        if (state[vIdx] == 1) return true;
        if (state[vIdx] == 0 && hasCycleDFS(vIdx, state))
            return true;
    }

    state[u] = 2;
    return false;
}
07

Topological Sort

نزور كل عقدة ونعمق في تابعاتها، ثم نضعها في المكدس. طباعة المكدس بالعكس تعطي الترتيب الصحيح.

void topoDFS(int u, bool visited[]) {
    visited[u] = true;

    for (int i = 0; i < adjCount[u]; i++) {
        int v = adj[u][i];
        int vIdx = courseIndex(v);
        if (vIdx != -1 && !visited[vIdx])
            topoDFS(vIdx, visited);
    }

    topoStack[++topoTop] = courses[u].id;
}
