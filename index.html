02

تمثيل الـ Graph

const int MAX = 100;

Course courses[MAX];
int courseCount = 0;

int prereq[MAX][MAX];
int prereqCount[MAX];

int adj[MAX][MAX];
int adjCount[MAX];

بدل ما نستخدم مكتبة جاهزة للـ Graph، بنسوي Adjacency List يدوياً باستخدام arrays:

• courses[]: مصفوفة المواد.
• adj[][]: كل مادة وموادها التابعة.
• prereq[][]: متطلبات كل مادة السابقة.
• courseCount: عدد المواد المخزنة.

إضافة مادة جديدة

void addCourse() {
    Course c;

    cout << "Enter course ID: ";
    cin >> c.id;

    if (courseIndex(c.id) != -1) {
        cout << "Course already exists!\n";
        return;
    }

    cout << "Enter course name: ";
    cin.ignore();
    getline(cin, c.name);

    cout << "Enter credit hours: ";
    cin >> c.hours;

    courses[courseCount] = c;
    courseCount++;

    cout << "Course added successfully.\n";
}

نطلب من المستخدم بيانات المادة، ثم نخزنها في المصفوفة في الموقع courseCount ونزيد العداد. قبل الإضافة نتحقق إن رقم المادة مو موجود من قبل.

    // التحقق من عدم وجود العلاقة مسبقاً
    for (int i = 0; i < prereqCount[cIdx]; i++) {

            cout << "Prerequisite already exists!\n";

لو المادة B تحتاج A، نضيف:

05

عرض المتطلبات (DFS)

void showPrerequisites(int id) {
    int idx = courseIndex(id);

    if (idx == -1)
        return;

    for (int i = 0; i < prereqCount[idx]; i++) {
        int preId = prereq[idx][i];

        cout << "- " << preId << "\n";

        // استدعاء recursive لعرض متطلبات المتطلب
        showPrerequisites(preId);
    }
}

نستخدم DFS (Depth First Search) عشان نعمق في المتطلبات: لو Algorithms يحتاج Data Structures، وData Structures يحتاج Programming 2، نطبعهم كلهم بالترتيب.

06

كشف الحلقات (Cycle Detection)

// state: 0 = ما زرته، 1 = قاعد أزوره، 2 = خلصت منه
bool hasCycleDFS(int u, int state[]) {
    state[u] = 1;

    for (int i = 0; i < adjCount[u]; i++) {
        int v = adj[u][i];
        int vIdx = courseIndex(v);

        if (vIdx == -1)
            continue;

        if (state[vIdx] == 1)
            return true;

        if (state[vIdx] == 0 && hasCycleDFS(vIdx, state))
            return true;
    }

    state[u] = 2;
    return false;
}

الحلقة تعني: مادة تعتمد على نفسها بشكل غير مباشر. نستخدم مصفوفة حالات بسيطة:
• 0 = ما زرناه
• 1 = قاعدين نزوره
• 2 = خلصنا منه
لو رجعنا لعقدة في الحالة 1، معناه حلقة.

07

Topological Sort (ترتيب المواد)

// مكدس للـ Topological Sort
int topoStack[MAX];
int topoTop = -1;

void topoDFS(int u, bool visited[]) {
    visited[u] = true;

    for (int i = 0; i < adjCount[u]; i++) {
        int v = adj[u][i];
        int vIdx = courseIndex(v);

        if (vIdx != -1 && !visited[vIdx]) {
            topoDFS(vIdx, visited);
        }
    }

    topoStack[++topoTop] = courses[u].id;
}

void topologicalSort() {
    bool visited[MAX] = {false};
    topoTop = -1;

    for (int i = 0; i < courseCount; i++) {
        if (!visited[i]) {
            topoDFS(i, visited);
        }
    }

    // طباعة المكدس بالعكس
    for (int i = topoTop; i >= 0; i--) {
        int idx = courseIndex(topoStack[i]);
        cout << courses[idx].id << " - "
             << courses[idx].name << "\n";
    }
}

DFS-based Topological Sort: نزور كل مادة ونعمق في تابعاتها، وبعد ما نخلص منها نضعها في المكدس. لما نطبع المكدس بالعكس، نحصل على الترتيب الصحيح لدراسة المواد.
